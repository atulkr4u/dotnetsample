Basic Query:
query {
  customers(searchCustomer: { customerId: 50 }) {
    customerId
    customerName
    pincode
    address
    city
  }
  orders(searchOrder: { customerId: 50 }) {
    orderId
    customerId
    productName
    price
    orderDate
  }
}

--Add Paging
query {
  customers(
    searchCustomer: { customerId: 5 }
  ) {
    nodes {
      customerId
      customerName
      pincode
      address
      city
    }
  }
  orders(
    searchOrder: { customerId: 5 }
    first: 5
  ) {
    pageInfo {
      hasNextPage
      hasPreviousPage
      startCursor
      endCursor
    }
    edges {
      cursor
      node {
        orderId
        customerId
        productName
        price
        orderDate
      }
    }
  }
}
